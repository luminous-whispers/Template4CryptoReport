\Require 输入：$x, y$
\Ensure 输出：$z$
\\
\Function{SampleProcedure}{$x, y$}
    \State $z \gets x + y$
    \If{$z > 10$}
        \State $z \gets z - 10$
    \ElsIf {$z<0$}
        \State \Return $None$
    \Else
        \State $hello$
    \EndIf
    \While{$z < 20$}
        \State $z \gets z + 1$
    \EndWhile
    \For{$i \gets 1$ to $10$}
        \State $z \gets z - i$
    \EndFor
    \State \Return $z$
\EndFunction
\\
